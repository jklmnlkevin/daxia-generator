<script>
var ${model}_toolbar = new Array();
// 新增按钮
<sec:authorize ifAnyGranted="${model}.add">
${model}_toolbar.push(k_toolbar_add('${model}'));
${model}_toolbar.push('-');
</sec:authorize>

//删除按钮
<sec:authorize ifAnyGranted="${model}.delete">
${model}_toolbar.push(k_toolbar_delete('${model}'));
${model}_toolbar.push('-');
</sec:authorize>

<sec:authorize ifAnyGranted="${model}.update">
// 修改按钮
${model}_toolbar.push({
    text:'修改',
    iconCls:'icon-edit',
    handler:function(){
    	var ids = k_getSelectedIds('${model}');
    	console.log('ids: ' + ids);
    	if (!!!ids || ids == '') {
    		alert2('请选择要修改的记录');
    		return;
    	} else if (ids.indexOf(',') != -1) {
    		alert2('只能选择一条记录进行修改');
    		return;
    	}
    	
    	$('#dd').dialog({
            title: '新增记录',
            width: 600,
            height: 300,
            closed: false,
            cache: false,
            href: '${'$'}{ctx}/admin/${model}/detail?id=' + ids,
            modal: true,
            buttons: [{
                text:'保存',
                iconCls:'icon-ok',
                handler:function(){
                    $('#${model}_detail_form').form('submit');
                }
            },{
                text:'取消',
                handler:function(){
                    $('#dd').dialog('close');
                }
            }]

        });
    }
});
</sec:authorize>

// 初始化 datagrid
function init() {
	$('#${model}_datagrid').datagrid({
	    url:'${'$'}{ctx}/admin/${model}/datagrid?1=1&',
	    method: "POST",
	    queryParams: {
			title: $('#title').val()
		},
	    columns:[[ // field是与json字段对应的，title是显示在表头的
	        {field:'ck', checkbox:'true'},
	        {field:'id',title:'ID',width:50},
	        <#list mf.shortComments as c>
				<#if c != 'id'>
		        {field:'${mf.names[c_index]}',title:'${c}',width:100,align:'center'}<#if c_has_next>,</#if>
				</#if>	
			</#list>
	    ]],
	    toolbar: ${model}_toolbar,
	    rownumbers: false,
	    pagination: true,
	    onLoadError: function() {},
	    onLoadSuccess: function(data) {}, 
	    onBeforeRender:function(target, rows) {alert('r	')},
	    loadFilter: function(data){
			if (data.statusCode && data.message){
				k_success(data);
			} else {
				return data;
			}
		}
	});
}

// 初始化datagrid
init();
